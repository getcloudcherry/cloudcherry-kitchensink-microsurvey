SDK has 6 events that can be listened on for capturing analytics data Replace token with the actual token number.

onImpression

Fired as soon as the survey is triggered

document.addEventListener(‘XX-XXX-ready', function(e){
  e.detail.survey.on('onImpression', function(e) {
   console.log('Impression');
  });
});

onQuestion

Fired when a question is visible on screen

document.addEventListener(‘XX-XXX-ready', function(e){
  e.detail.survey.on('onQuestion', function(e) {
   console.log('Question Displayed');
  });
});

onAnswer

Fired when a question is answered

document.addEventListener(‘XX-XXX-ready', function(e){
  e.detail.survey.on('onAnswer', function(e) {
   console.log('Question Answered');
  });
});

onClick

Fired every time a click event happen on survey

document.addEventListener(‘XX-XXX-ready', function(e){
  e.detail.survey.on('onClick', function(e) {
   console.log('Clicked');
  });
});

onClose

Fired when user closes the survey

document.addEventListener(‘XX-XXX-ready', function(e){
  e.detail.survey.on('onClose', function(e) {
   console.log('Closed');
  });
});

onEnd

Fired when the survey ends

document.addEventListener(‘XX-XXX-ready', function(e){
  e.detail.survey.on('onEnd', function(e) {
   console.log('Survey Ended');
  });
});
